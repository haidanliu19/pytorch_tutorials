class DNN(nn.Module):
    def __init__(self, num_classes=10):
        super(DNN, self).__init__()
        self.layer1 = nn.Sequential(
            ##   ##
            nn.Linear(28 * 28, 512),    # Linear_1  
            nn.BatchNorm1d(512),    # BatchNorm_1  
            nn.ReLU(),     # ReLU_1  
            ##   ##
        )
        self.layer2 = nn.Sequential(
            ##   ##
            nn.Linear(512, 10),    # Linear_2   
            ##   ##
        )
    
    def forward(self, x):
        x = x.view(x.size(0), -1) # flatten
        x_out = self.layer1(x)
        x_out = self.layer2(x_out)
        return x_out

for epoch in range(num_epochs):
    for i, (imgs, labels) in enumerate(train_loader):
        imgs, labels = imgs.to(device), labels.to(device)
        ##   ##
        outputs = model(imgs)  #   1.   None .
        loss = criterion(outputs, labels)     #   2.   None .
        
        optimizer.zero_grad()            # Clear gradients:   3.   None .
        loss.backward()            # Gradients :   4.   None .
        optimizer.step()            # Parameters :   5.   None .
        ##   ##
        
        _, argmax = torch.max(outputs, 1)
        accuracy = (labels == argmax).float().mean()
        
        if (i+1) % 100 == 0:
            print('Epoch [{}/{}], Step [{}/{}], Loss: {:.4f}, Accuracy: {:.2f}%'.format(
                epoch+1, num_epochs, i+1, len(train_loader), loss.item(), accuracy.item() * 100))
